-- Create the database if it doesn't exist
IF NOT EXISTS (SELECT * FROM sys.databases WHERE name = 'resume_portal')
BEGIN
    CREATE DATABASE resume_portal;
END
GO

USE resume_portal;
GO

-- To store job postings
CREATE TABLE resume_jobs (
    job_id NVARCHAR(255) PRIMARY KEY,
    jobTitle NVARCHAR(255) NOT NULL,
    department NVARCHAR(255),
    location NVARCHAR(255),
    workType NVARCHAR(50),
    workMode NVARCHAR(50),
    experienceLevel NVARCHAR(100),
    minExperience INT,
    maxExperience INT,
    band NVARCHAR(255),
    preferredIndustry NVARCHAR(255),
    minSalary BIGINT,
    maxSalary BIGINT,
    currency NVARCHAR(10),
    jobDescription NVARCHAR(MAX),
    responsibilities NVARCHAR(MAX) CHECK (ISJSON(responsibilities) > 0),
    requirements NVARCHAR(MAX) CHECK (ISJSON(requirements) > 0),
    qualifications NVARCHAR(MAX) CHECK (ISJSON(qualifications) > 0),
    skills NVARCHAR(MAX) CHECK (ISJSON(skills) > 0),
    benefits NVARCHAR(MAX) CHECK (ISJSON(benefits) > 0),
    applicationDeadline DATE,
    positionsAvailable INT DEFAULT 1,
    reportingTo NVARCHAR(255),
    contactEmail NVARCHAR(255),
    isUrgent BIT DEFAULT 0,
    allowRemote BIT DEFAULT 0,
    travelRequired BIT DEFAULT 0,
    backgroundCheckRequired BIT DEFAULT 0,
    postedDate DATETIME DEFAULT GETDATE(),
    status NVARCHAR(50) DEFAULT 'Active'
);
GO

-- To store candidate information and resume metadata
CREATE TABLE resume_candidates (
    resume_id NVARCHAR(36) PRIMARY KEY,
    job_id NVARCHAR(255),
    first_name NVARCHAR(255),
    last_name NVARCHAR(255),
    email NVARCHAR(255) NOT NULL,
    phone NVARCHAR(20),
    gender NVARCHAR(20),
    age INT,
    work_pref NVARCHAR(100),
    address NVARCHAR(MAX),
    experience NVARCHAR(100),
    marks12 NVARCHAR(50),
    pass12 INT,
    grad_year INT,
    grad_marks NVARCHAR(50),
    linkedin NVARCHAR(255),
    s3_key NVARCHAR(255),
    resume_url NVARCHAR(MAX),
    status NVARCHAR(50) DEFAULT 'Uploaded',
    submission_timestamp DATETIME,
    extracted_text NVARCHAR(MAX),
    extracted_entities NVARCHAR(MAX) CHECK (ISJSON(extracted_entities) > 0),
    extracted_skills NVARCHAR(MAX) CHECK (ISJSON(extracted_skills) > 0),
    FOREIGN KEY (job_id) REFERENCES resume_jobs(job_id) ON DELETE SET NULL
);
GO

-- To store the secure review tokens
CREATE TABLE resume_review_tokens (
    token NVARCHAR(512) PRIMARY KEY,
    resume_id NVARCHAR(36) NOT NULL,
    status NVARCHAR(20) DEFAULT 'pending',
    expires_at DATETIME NOT NULL,
    FOREIGN KEY (resume_id) REFERENCES resume_candidates(resume_id) ON DELETE CASCADE
);
GO

-- Drop the table if it already exists
IF OBJECT_ID('dbo.resume_employees', 'U') IS NOT NULL
    DROP TABLE dbo.employees;
GO

-- Create the new employees table
CREATE TABLE resume_employees (
    employee_id INT PRIMARY KEY IDENTITY(1,1),
    name NVARCHAR(255) NOT NULL,
    email NVARCHAR(255) NOT NULL UNIQUE,
    department NVARCHAR(100) NOT NULL
);
GO
